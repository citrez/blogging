overall_avg <- mpg %>% summarise(mean(cty)) %>% pull()

(g <- mpg %>% 
  ggplot(aes(cty,hwy,color = class))+
  geom_abline(slope = 0,intercept  = overall_avg,color = 'grey70',size = 1)+
  geom_jitter(position = position_jitter(seed= 2021,height = 20) )+ # didnt know you cuodl set seed here
  scale_x_continuous(expand = c(0.05,0.05))+
    geom_rug(show.legend = F,sides = "tr",size=3,alpha = 0.3)+ # use geom rug, v easy, v cool
    ggsci::scale_color_rickandmorty()+
  labs(x = "City (mpg)",title = "Exapmple of how i might use fonts")+
  theme(title = element_text(family = "Rubik Mono One"),
        legend.text =  element_text(family = "gochi"),
        panel.grid    = element_blank()
        )+
  stat_summary(fun = median,  geom = 'point',color = 'white',size = 3,alpha = 0.6) )

g_for_grob <- g+
    theme(line = element_blank(),
